Another API to learn ?

Symfony BrowserKit


$client = static::createClient();

$client->request('GET', '/coffreo');

$crawler = $client->clickLink('Create new');

$this->assertContains('Create new Coffreo', $crawler->html());

Symfony Panther


$client = static::createPantherClient();

$client->request('GET', '/coffreo');

$crawler = $client->clickLink('Create new');

$this->assertContains('Create new Coffreo', $crawler->html());
